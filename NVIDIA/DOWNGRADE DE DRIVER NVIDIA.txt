DOWNGRADE DE DRIVER NVIDIA

# LISTA DRIVERS INSTALADOS:

apt list '*nvidia-driver*'

# APAGUE AS CHAVES GPG:

/usr/share/keyrings/nvidia-drivers.gpg

/etc/apt/sources.list.d/nvidia-drivers.list


# REMOVE TUDO (Comprovado)

sudo apt-get purge '*nvidia*'

# ESCOLHA UM DRIVER E INSTALE :

Ex;
sudo apt install nvidia-driver-470-server

sudo service gdm3 restart
